--- Callbacks 回调对象

function fn1() {alert(1)}
function fn2() {alert(2)}

var cb = $.Callbacks();

cb.add(fn1);
cb.add(fn2);

cb.fire() // 1, 2

cb.remove(fn2)

cb.fire() // 1

--- Deferred 延迟对象

setTimeout(function() {
  alert(1)
},1000)

alert(2)


var dfd = $.Deferred();

setTimeout(function() {
  alert(1);

  dfd.resolve();  
},1000)

dfd.done(function() {
  alert(2);
})
